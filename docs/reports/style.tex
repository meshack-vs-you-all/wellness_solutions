% Progress box macro for module status
\newcommand{\progressbox}[3]{%
    \begin{tcolorbox}[
        enhanced,
        boxrule=0.5pt,
        colback=#3!10,
        colframe=#3!50,
        arc=1mm,
        width=\linewidth
    ]
        \textbf{#1}\\[0.1cm]
        \begin{tikzpicture}[x=1pt,y=1pt]
            \fill[#3!20] (0,0) rectangle (\linewidth,8);
            \fill[#3] (0,0) rectangle ({#2*\linewidth/100},8);
        \end{tikzpicture}\\[-0.1cm]
        \hfill#2\%
    \end{tcolorbox}%
}
