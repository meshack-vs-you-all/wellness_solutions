class BookingCreateView(LoginRequiredMixin, CreateView):
    model = Booking
    form_class = BookingForm
    template_name = 'bookings/create.html'
    
    def form_valid(self, form):
        # Set the client to current user
        form.instance.client = self.request.user
        
        try:
            # Check for scheduling conflicts
            if form.instance.has_conflicts():
                form.add_error(
                    None, 
                    'Time slot is not available'
                )
                return self.form_invalid(form)
            
            # Create the booking
            response = super().form_valid(form)
            
            # Send confirmation email
            send_booking_confirmation.delay(
                booking_id=self.object.id
            )
            
            return response
            
        except Exception as e:
            logger.error(
                f"Booking creation failed: {str(e)}"
            )
            form.add_error(
                None,
                'Unable to process booking'
            )
            return self.form_invalid(form)
    
    def get_success_url(self):
        return reverse_lazy(
            'booking-detail',
            kwargs={'pk': self.object.pk}
        )
    \end{verbatim}
\end{tcolorbox}

class BookingModelTest(TestCase):
    def setUp(self):
        self.user = User.objects.create_user(
            username='testuser',
            email='test@example.com',
            password='testpass123'
        )
        self.instructor = User.objects.create_user(
            username='instructor',
            email='instructor@example.com',
            password='instructor123'
        )

    def test_booking_creation(self):
        booking = Booking.objects.create(
            client=self.user,
            instructor=self.instructor,
            start_time=timezone.now(),
            end_time=timezone.now() + timedelta(hours=1)
        )
        self.assertTrue(isinstance(booking, Booking))
        self.assertEqual(booking.client, self.user)

    def test_booking_validation(self):
        with self.assertRaises(ValidationError):
            booking = Booking(
                client=self.user,
                start_time=timezone.now(),
                end_time=timezone.now() - timedelta(hours=1)
            )
            booking.full_clean()
