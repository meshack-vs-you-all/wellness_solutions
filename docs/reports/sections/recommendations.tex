\section{Technical Debt Analysis}

\begin{tcolorbox}[
    enhanced,
    title={Code Quality Issues},
    attach boxed title to top left,
    boxed title style={
        colback=warning!50,

        arc=0mm,

    colback=white,
    colframe=warning!50,
    arc=0mm,
    fonttitle=\bfseries
]
    \subsection{Validation Logic}
    Current implementation:
    \begin{lstlisting}[language=Python]
    def validate_booking(self):
        if self.end_time <= self.start_time:
            raise ValidationError("Invalid time range")
        if self.client == self.instructor:
            raise ValidationError("Invalid assignment")
    \end{lstlisting}
    
    Recommended approach:
    \begin{lstlisting}[language=Python]
    def validate_booking(self):
        self.validate_time_range()
        self.validate_participants()
        self.validate_availability()
        self.validate_package_status()
    
    def validate_time_range(self):
        if self.end_time <= self.start_time:
            raise ValidationError({
                "time_range": "End time must be after start time"
            })
    \end{lstlisting}

    \subsection{Query Optimization}
    Current pattern:
    \begin{lstlisting}[language=Python]
    bookings = Booking.objects.filter(date=today)
    for booking in bookings:
        client = booking.client  # N+1 query problem
        instructor = booking.instructor
    \end{lstlisting}
    
    Optimized approach:
    \begin{lstlisting}[language=Python]
    bookings = Booking.objects.select_related(
        'client', 'instructor'
    ).filter(date=today)
    \end{lstlisting}
\end{tcolorbox}
